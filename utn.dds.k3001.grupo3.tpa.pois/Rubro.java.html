public class Rubro 
{

		this.nombre = nombre;
		this.distanciaDeCercania = distanciaDeCercania;
	}
	
	public double distanciaDeCercania() {
		return distanciaDeCercania;
	}
	
	public String nombre() {
		return nombre;
	}
}
